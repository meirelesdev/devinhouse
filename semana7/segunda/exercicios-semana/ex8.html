Exercicio 8

No exercício 6, você criou uma função que retornava uma promise que resolvia, após 3 segundos, com um objeto converta esta função para a forma async / await.

Resposta:

        function getAuth() {
            return new Promise((resolve) => {
                setTimeout(() => {
                    resolve({username: "Daniel",email: "daniel_m_nascimento@estudante.sesisenai.org.br",authorized: true})
                }, 3000);
            });
        }
        async function showResult(res){
            const divResponse = document.querySelector('.response')
            divResponse.innerHTML = 'Consultando aguarde...'
            const credentials =  await getAuth()
            divResponse.innerHTML = JSON.stringify(credentials)
            console.log(credentials)
        }
        showResult()
