Ex 5

Uma classe agrupa características e comportamentos de algo, que pode ser um objeto palpável ou algo abstrato. Crie 2 classes que representam objetos reais e duas classes que representem objetos abstratos, cada uma com pelo menos 2 atributos, que serão recebidos em seus respectivos métodos construtores durante a instanciação.

Classes de Objetos reais

                class Bola {
                    #cor
                    #tipo
                    constructor(cor, tipo) {
                        this.#cor = cor
                        this.#tipo = tipo
                    }
                    toString(){
                        return `Bola cor: ${this.#cor} do tipo: ${this.#tipo}`
                    }
                }
                class Veiculo {
                    #modelo
                    #cor
                    constructor(modelo, cor, ) {
                        this.#modelo = modelo
                        this.#cor = cor
                    }
                    toString(){
                        return `Modelo: ${this.#modelo}, Cor: ${this.#cor}`
                    }
                }

Classes de Objetos Abstratos

                class Clima {
                    #temperatura
                    #temSol
                    constructor(temperatura, temSol){
                        this.#temperatura = temperatura
                        this.#temSol = temSol
                    }
                    toString(){
                        return `A temperatura esta em: ${this.#temperatura} e o Sol esta ${this.#temSol ? 'Lindo': 'Escondigo'}.`
                    }
                }
                class Biografia {
                    #dono
                    #descricao
                    constructor(dono, descricao){
                        this.#dono = dono
                        this.#descricao = descricao
                    }
                    toString(){
                        return `Autor: ${this.#dono} Descrição: ${this.#descricao}`
                    }
                }
